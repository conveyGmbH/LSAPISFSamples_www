ðŸ’» Basic Usage

// 1. Include the library
<script src="salesforceLeadLib.js"></script>

// 2. Initialize the module
const container = document.getElementById('salesforceContainer');

const leadData = {
    FirstName: 'John',
    LastName: 'Doe',
    Company: 'Acme Corporation',
    Email: 'john.doe@acme.com',
    Phone: '+1 555-0123',
    Title: 'Marketing Director',
    EventId: '12345'
};

const credentials = {
    username: 'demo@leadsuccess.com',
    orgId: '00D5g000000abcd',
    backendUrl: 'http://localhost:3000'
};

const options = {
    title: 'Salesforce Lead Transfer',
    eventId: '12345',
    onDisconnect: () => {
        console.log('User clicked disconnect');
    },
    onDisconnected: () => {
        console.log('Module cleared after disconnect');
    }
};

// Initialize (returns Promise or WinJS.Promise)
SalesforceLeadLib.initialize(container, leadData, credentials, options)
    .then(result => {
        console.log('Module initialized:', result);
    })
    .catch(error => {
        console.error('Initialization failed:', error);
    });

ðŸŽ¯ WinJS Integration (LSPortal)

// In LSPortal (WinJS application)
(function () {
    "use strict";

    WinJS.UI.Pages.define("/pages/salesforceLead/salesforceLead.html", {
        ready: function (element, options) {
            const container = element.querySelector('#sfContainer');

            // Get lead data from your data source
            const leadData = AppData.getCurrentLead();

            const credentials = {
                username: AppData.currentUser.email,
                orgId: AppData.salesforceOrgId,
                backendUrl: 'https://your-backend.com'
            };

            // Initialize the module (returns WinJS.Promise automatically)
            SalesforceLeadLib.initialize(container, leadData, credentials, {
                eventId: leadData.EventId,
                onDisconnect: () => {
                    // Handle disconnect
                    WinJS.Navigation.back();
                }
            }).then(
                function (result) {
                    console.log('SF Module ready:', result);
                },
                function (error) {
                    console.error('SF Module failed:', error);
                    // Show error page
                }
            );
        },

        unload: function () {
            // Cleanup when page unloads
            const container = document.querySelector('#sfContainer');
            if (container) {
                SalesforceLeadLib.clear(container);
            }
        }
    });
})();
